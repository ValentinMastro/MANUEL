\usepackage{import}
\newcommand{\exercice}[2]{
    % #1: numéro de l'exercice
    % #2: chemin relatif du fichier contenant l'exercice
    \section*{Exercice #1}
    \addcontentsline{toc}{section}{Exercice #1}
    \import{#2}{#1}
}

\newcommand{\importerExercices}[2]{
    % #1: numéro du dernier exercice
    % #2: chemin relatif du répertoire contenant les exercices
    \foreach \n in {1, ..., #1} {
        \exercice{\ifnum \n < 100 0\fi\ifnum \n < 10 0\fi\n}{#2}
    }
}
